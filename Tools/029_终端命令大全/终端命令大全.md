

# Xcode

```sh
# 移除现有的 CLT
$ sudo rm -rf /Library/Developer/CommandLineTools

# 安装
$ sudo xcode-select --install

# 验证
$ xcode-select -p
```





# brew

```sh
# 使用 Homebrew 安装 MySQL
$ brew install mysql
# 具体信息
$ brew info mysql


# 使用 Homebrew 管理（推荐）
$ brew services start mysql
$ brew services stop mysql
$ brew services restart mysql

# 停止当前 MySQL 服务
$ mysql.server start
$ mysql.server stop


# 检查服务状态
$ brew services list
$ mysql.server status

# 更新包
$ brew upgrade mysql

# 卸载
$ brew remove mysql
$ brew cleanup
```
